1112 //-------------------------------------------------------------------------
1113 int MainFrame::rxCallback(
1114  const void *inputBuffer,
1115  void *outputBuffer,
1116  unsigned long framesPerBuffer,
1117  const PaStreamCallbackTimeInfo* timeInfo,
1118  PaStreamCallbackFlags statusFlags,
1119  void *userData
1120  )
1121 {
1122  paCallBackData *cbData = (paCallBackData*)userData;
1123  unsigned int i;
1124  short *rptr = (short*)inputBuffer;
1125  short *wptr = (short*)outputBuffer;
1126  float *in8k = cbData->in8k;
1127  float *in48k = cbData->in48k;
1128  float out8k[N8];
1129  float out48k[N48];
1130  short out48k_short[N48];
1131  short in48k_short[N48];
1132  short indata[MAX_FPB];
1133  short outdata[MAX_FPB];
1134 
1135  (void) timeInfo;
1136  (void) statusFlags;
1137 
1138  assert(inputBuffer != NULL);
1139  assert(outputBuffer != NULL);
1140  /*
1141  framesPerBuffer is portaudio-speak for number of samples we
1142  actually get from the record side and need to provide to the
1143  play side. On Linux (at least) it was found that
1144  framesPerBuffer may not always be what we ask for in the
1145  framesPerBuffer field of Pa_OpenStream. For example a request
1146  for 960 sample buffers lead to framesPerBuffer = 1024.
1147 
1148  To perform the 48 to 8 kHz conversion we need an integer
1149  multiple of FDMDV_OS samples to support the interpolation and
1150  decimation. As we can't guarantee the size of framesPerBuffer
1151  we do a little FIFO buffering.
1152  */
1153 
1154  /* assemble a mono buffer (just use left channel) and write to FIFO */
1155 
1156  assert(framesPerBuffer < MAX_FPB);
1157  for(i=0; i<framesPerBuffer; i++,rptr+=2)
1158  indata[i] = *rptr;
1159  fifo_write(cbData->infifo, indata, framesPerBuffer);
1160 
1161  /* while we have enough samples available ... */
1162 
1163  while (fifo_read(cbData->infifo, in48k_short, N48) == 0) {
1164 
1165  /* convert to float */
1166 
1167  for(i=0; i<N48; i++)
1168  in48k[FDMDV_OS_TAPS + i] = in48k_short[i];
1169 
1170  // downsample and update filter memory
1171  fdmdv_48_to_8(out8k, &in48k[FDMDV_OS_TAPS], N8);
1172  for(i = 0; i < FDMDV_OS_TAPS; i++)
1173  {
1174  in48k[i] = in48k[i + N48];
1175  }
1176 
1177  assert((g_nInputBuf + N8) <= 2 * FDMDV_NOM_SAMPLES_PER_FRAME);
1178  // run demod, decoder and update GUI info
1179  for(i = 0; i < N8; i++)
1180  {
1181  g_RxInBuf[g_nInputBuf + i] = (short)out8k[i];
1182  }
1183  g_nInputBuf += FDMDV_NOM_SAMPLES_PER_FRAME;
1184  per_frame_rx_processing(g_pRxOutBuf, &g_nOutputBuf, g_CodecBits, g_RxInBuf, &g_nInputBuf, &g_nRxIn, &g_State, g_pCodec2);
1185  //cbData->pWFPanel->m_newdata = true;
1186  //cbData->pSPPanel->m_newdata = true;
1187 
1188  // if demod out of sync copy input audio from A/D to aid in tuning
1189  if (g_nOutputBuf >= N8)
1190  {
1191  if(g_State == 0)
1192  {
1193  for(i = 0; i < N8; i++)
1194  {
1195  in8k[MEM8 + i] = out8k[i]; // A/D signal
1196  }
1197  }
1198  else
1199  {
1200  for(i = 0; i < N8; i++)
1201  {
1202  in8k[MEM8+i] = g_pRxOutBuf[i]; // decoded spech
1203  }
1204  }
1205  g_nOutputBuf -= N8;
1206  }
1207  assert(g_nOutputBuf >= 0);
1208  // shift speech samples in output buffer
1209  for(i = 0; i < (unsigned int)g_nOutputBuf; i++)
1210  {
1211  g_pRxOutBuf[i] = g_pRxOutBuf[i + N8];
1212  }
1213 
1214  /* test: echo input to output, make this loopback option */
1215  for(i=0; i<N8; i++)
1216  in8k[MEM8+i] = out8k[i];
1217 
1218  // Convert output speech to 48 kHz sample rate
1219  // upsample and update filter memory
1220  fdmdv_8_to_48(out48k, &in8k[MEM8], N8);
1221  for(i = 0; i < MEM8; i++)
1222  {
1223  in8k[i] = in8k[i + N8];
1224  }
1225  assert(outputBuffer != NULL);
1226 
1227  // write signal to fifo
1228  for(i = 0; i < N48; i++)
1229  {
1230  out48k_short[i] = (short)out48k[i];
1231  }
1232 
1233  fifo_write(cbData->outfifo, out48k_short, N48);
1234  }
1235 
1236  /* OK now set up output samples */
1237 
1238  if (fifo_read(cbData->outfifo, outdata, framesPerBuffer) == 0) {
1239

1241 
1242  for(i=0; i<framesPerBuffer; i++,wptr+=2) {
1243  wptr[0] = outdata[i];
1244  wptr[1] = outdata[i];
1245  }
1246  }
1247  else {
1248  //printf("no data\n");
1249  /* zero output if no data available */
1250  for(i=0; i<framesPerBuffer; i++,wptr+=2) {

1252  wptr[1] = 0;
1253  }
1254  }

1259 //----------------------------------------------------------------
1260 // per_frame_rx_processing()
1261 //----------------------------------------------------------------
1262  void MainFrame::per_frame_rx_processing(
1263  short output_buf[], // output buf of decoded speech samples
1264  int *n_output_buf, // how many samples currently in output_buf[]
1265  int codec_bits[], // current frame of bits for decoder
1266  short input_buf[], // input buf of modem samples input to demod
1267  int *n_input_buf, // how many samples currently in input_buf[]
1268  int *nin, // amount of samples demod needs for next call
1269  int *state, // used to collect codec_bits[] halves
1270  CODEC2 *c2 // Codec 2 states
1271  )
1272 {
1273  struct FDMDV_STATS stats;
1274  int sync_bit;
1275  float rx_fdm[FDMDV_MAX_SAMPLES_PER_FRAME];
1276  int rx_bits[FDMDV_BITS_PER_FRAME];
1277  unsigned char packed_bits[BYTES_PER_CODEC_FRAME];
1278  float rx_spec[FDMDV_NSPEC];
1279  int i;
1280  int nin_prev;
1281  int bit;
1282  int byte;
1283  int next_state;
1284 
1285  assert(*n_input_buf <= (2 * FDMDV_NOM_SAMPLES_PER_FRAME));
1286 
1287  //
1288  // This while loop will run the demod 0, 1 (nominal) or 2 times:
1289  //
1290  // 0: when tx sample clock runs faster than rx, occasionally we
1291  // will run out of samples
1292  //
1293  // 1: normal, run decoder once, every 2nd frame output a frame of
1294  // speech samples to D/A
1295  //
1296  // 2: when tx sample clock runs slower than rx, occasionally we will
1297  // have enough samples to run demod twice.
1298  //
1299  // With a +/- 10 Hz sample clock difference at FS=8000Hz (+/- 1250
1300  // ppm), case 0 or 1 occured about once every 30 seconds. This is
1301  // no problem for the decoded audio.
1302  //
1303  while(*n_input_buf >= *nin)
1304  {
1305  // demod per frame processing
1306  for(i = 0; i < *nin; i++)
1307  {
1308  rx_fdm[i] = (float)input_buf[i] / FDMDV_SCALE;
1309  }
1310  nin_prev = *nin;
1311  fdmdv_demod(g_pFDMDV, rx_bits, &sync_bit, rx_fdm, nin);
1312  *n_input_buf -= nin_prev;
1313  assert(*n_input_buf >= 0);
1314 
1315  // shift input buffer
1316  for(i = 0; i < *n_input_buf; i++)
1317  {
1318  input_buf[i] = input_buf[i + nin_prev];
1319  }
1320 
1321  // compute rx spectrum & get demod stats, and update GUI plot data
1322  fdmdv_get_rx_spectrum(g_pFDMDV, rx_spec, rx_fdm, nin_prev);
1323  fdmdv_get_demod_stats(g_pFDMDV, &stats);
1324  // Average Data
1325  // averageData(rx_spec);
1326  for(i = 0; i < FDMDV_NSPEC; i++)
1327  {
1328  // m_rxPa->m_av_mag[i] = (1.0 - BETA) * m_rxPa->m_av_mag[i] + BETA * rx_spec[i];
1329  g_avmag[i] = (1.0 - BETA) * g_avmag[i] + BETA * rx_spec[i];
1330  }
1331  //
1332  // State machine to:
1333  //
1334  // + Mute decoded audio when out of sync. The demod is synced
1335  // when we are using the fine freq estimate and SNR is above
1336  // a thresh.
1337  //
1338  // + Decode codec bits only if we have a 0,1 sync bit
1339  // sequence. Collects two frames of demod bits to decode
1340  // one frame of codec bits.
1341  //
1342  next_state = *state;
1343  switch(*state)
1344  {
1345  case 0:
1346  // mute output audio when out of sync
1347  if(*n_output_buf < 2 * codec2_samples_per_frame(c2) - N8)
1348  {
1349  for(i = 0; i < N8; i++)
1350  {
1351  output_buf[*n_output_buf + i] = 0;
1352  }
1353  *n_output_buf += N8;
