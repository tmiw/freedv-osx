65 void AudioDlg::ExchangeData(int inout)
66 {
67  wxConfigBase *pConfig = wxConfigBase::Get();
68  if(inout == EXCHANGE_DATA_IN)
69  {
70  m_textRxInput->SetValue(wxGetApp().m_strRxInAudio);
71  m_textTxOutput->SetValue(wxGetApp().m_strRxOutAudio);
72  m_textVoiceInput->SetValue(wxGetApp().m_textVoiceInput);
73  m_textVoiceOutput->SetValue(wxGetApp().m_textVoiceOutput);
74  }
75  if(inout == EXCHANGE_DATA_OUT)
76  {
77  wxGetApp().m_strRxInAudio = m_textRxInput->GetValue();
78  wxGetApp().m_strRxOutAudio = m_textTxOutput->GetValue();
79  wxGetApp().m_textVoiceInput = m_textVoiceInput->GetValue();
80  wxGetApp().m_textVoiceOutput = m_textVoiceOutput->GetValue();
81 
82  pConfig->Write(wxT("/Audio/RxIn"), wxGetApp().m_strRxInAudio);
83  pConfig->Write(wxT("/Audio/RxOut"), wxGetApp().m_strRxOutAudio);
84  pConfig->Write(wxT("/Audio/TxIn"), wxGetApp().m_textVoiceInput);
85  pConfig->Write(wxT("/Audio/TxOut"), wxGetApp().m_textVoiceOutput);
86  pConfig->Write(wxT("/Audio/SampleRate"), wxGetApp().m_strSampleRate);
87  pConfig->Flush();
88  }
89  delete wxConfigBase::Set((wxConfigBase *) NULL);
90 }
