131 void PlotSpectrum::drawGraticule(wxAutoBufferedPaintDC& dc)
132 {
133  int p;
134  char buf[15];
135  wxString s;

138  // Vertical gridlines
139  dc.SetPen(m_penShortDash);
140  for(p = (PLOT_BORDER + XLEFT_OFFSET + GRID_INCREMENT); p < ((m_rGrid.GetWidth() - XLEFT_OFFSET) + GRID_INCREMENT); p += GRID_INCREMENT)

144 
145  int y_zero = (m_rGrid.GetHeight() - m_top) / 2 ;

170  {
171  sprintf(buf, "%1.1f Hz",(double)(p / 10));
172  dc.DrawText(buf, p - PLOT_BORDER + XLEFT_OFFSET, m_rCtrl.GetHeight() + YBOTTOM_OFFSET/2);
