184 void PlotPanel::draw(wxAutoBufferedPaintDC& pDC)
185 {
186  wxMemoryDC m_mDC;
187  m_mDC.SelectObject(*m_pBmp);
188  m_rCtrl = GetClientRect();
189  m_rGrid = m_rCtrl;
190 
191  m_rGrid = m_rGrid.Deflate(PLOT_BORDER + (XLEFT_OFFSET/2), (PLOT_BORDER + (YBOTTOM_OFFSET/2)));
192  m_rGrid.Offset(PLOT_BORDER + XLEFT_OFFSET, PLOT_BORDER);
193 
194  pDC.Clear();
195  m_rPlot = wxRect(PLOT_BORDER + XLEFT_OFFSET, PLOT_BORDER, m_rGrid.GetWidth(), m_rGrid.GetHeight());
196  if(m_firstPass)
197  {
198  m_firstPass = false;
199  m_mDC.FloodFill(0, 0, VERY_LTGREY_COLOR);
200 
201  // Draw a filled rectangle with aborder
202  wxBrush ltGraphBkgBrush = wxBrush(DARK_BLUE_COLOR);
203  m_mDC.SetBrush(ltGraphBkgBrush);
204  m_mDC.SetPen(wxPen(BLACK_COLOR, 0));
205  m_mDC.DrawRectangle(m_rPlot);
206  }
207  if(m_newdata)
208  {
209  m_newdata = false;
210  int t = m_rPlot.GetTop();
211  int l = m_rPlot.GetLeft();
212 // int r = m_rPlot.GetRight();
213  int h = m_rPlot.GetHeight();
214  int w = m_rPlot.GetWidth();
215  pDC.Blit(l, t, w, h, &m_mDC, l, t);
216  }
217  drawGraticule(pDC);
218  m_mDC.SetBrush(wxNullBrush);
219  m_mDC.SelectObject(wxNullBitmap);
220 }

222 //-------------------------------------------------------------------------
223 // drawGraticule()
224 //-------------------------------------------------------------------------
225 void PlotPanel::drawGraticule(wxAutoBufferedPaintDC& pDC)
226 {
227  int p;
228  char buf[15];
229  wxString s;
230 
231  // Vertical gridlines
232  pDC.SetPen(m_penShortDash);
233  for(p = (PLOT_BORDER + XLEFT_OFFSET + GRID_INCREMENT); p < ((m_rGrid.GetWidth() - XLEFT_OFFSET) + GRID_INCREMENT); p += GRID_INCREMENT)
234  {
235  pDC.DrawLine(p, (m_rGrid.GetHeight() + PLOT_BORDER), p, PLOT_BORDER);
236  }
237  // Horizontal gridlines
238  pDC.SetPen(m_penDotDash);
239  for(p = (m_rGrid.GetHeight() - GRID_INCREMENT); p > PLOT_BORDER; p -= GRID_INCREMENT)
240  {
241  pDC.DrawLine(PLOT_BORDER + XLEFT_OFFSET, (p + PLOT_BORDER), (m_rGrid.GetWidth() + PLOT_BORDER + XLEFT_OFFSET), (p + PLOT_BORDER));
242  }
243  // Label the X-Axis
244  pDC.SetPen(wxPen(GREY_COLOR, 1));
245  for(p = GRID_INCREMENT; p < (m_rGrid.GetWidth() - YBOTTOM_OFFSET); p += GRID_INCREMENT)
246  {
247  sprintf(buf, "%1.1f Hz",(double)(p / 10));
248  pDC.DrawText(buf, p - PLOT_BORDER + XLEFT_OFFSET, m_rGrid.GetHeight() + YBOTTOM_OFFSET/2);
249  }
250  // Label the Y-Axis
251  //for(p = GRID_INCREMENT; p < (h - YBOTTOM_OFFSET); p += GRID_INCREMENT)
252  for(p = (m_rGrid.GetHeight() - GRID_INCREMENT); p > PLOT_BORDER; p -= GRID_INCREMENT)
253  {
254  sprintf(buf, "%1.0f", (double)((m_rGrid.GetHeight() - p) * -10));
255  pDC.DrawText(buf, XLEFT_TEXT_OFFSET, p);
256  }
