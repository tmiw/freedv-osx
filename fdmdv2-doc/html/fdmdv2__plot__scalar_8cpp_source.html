101 void PlotScalar::draw(wxAutoBufferedPaintDC& dc)
102 {

109  int y2;

113  /* detect resizing of window */

124  //fl_push_clip(m_x, m_y, m_w, m_h);
125  x_scale = (float)m_x_max;
126  y_scale = (float)m_rCtrl.GetHeight() /(2.0 * m_y_max);
127 
128  // erase last sample
129  dc.SetPen(BLACK_COLOR);
130  x1 = x_scale * m_index + m_x;
131  y1 = y_scale * m_mem[m_index];
132  y1 = clip(y1);

141  y1 = m_y + m_rCtrl.GetHeight()/2 - y1;
142  dc.DrawPoint(x1, y1);
143  m_mem[m_index] = m_new_sample;
144  m_index++;
145  if(m_index >= m_x_max)
146  {
147  m_index = 0;
148  }
149 
150  // y axis graticule

153  {

157  {
158  m_step /= 2.0;
159  }
160  pen = dc.GetPen();
161  pen.SetColour(DARK_GREEN_COLOR);
162  pen.SetStyle(wxPENSTYLE_DOT);

168  x2 = m_x + m_rCtrl.GetWidth();

171  }
172

179  {
180  x1 = m_x;
181  y1 = m_y + m_rCtrl.GetHeight()/2 - i * y_scale;
182  sprintf(label, "%d", i);
